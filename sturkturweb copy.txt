└── src/                           # Folder utama seluruh kode sumber aplikasi bot
    ├── __init__.py                # Berkas inisialisasi paket src
    ├── game_data.py               # Database/kamus tunggal berisi data statis (senjata, item, terrain, cuaca) serta peta global
    ├── config.py                  # Pengatur konfigurasi berbasis variabel lingkungan dari .env
    ├── api.py                     # Modul komunikasi HTTP API (otentikasi, klaim reward, dll)
    ├── game_log.py                # Modul pemformatan keluaran visual log turn di terminal secara rapi
    ├── state_manager.py           # Pengelola status state terintegrasi dari seluruh pesan WebSocket
    ├── agent_memory.py            # Modul pengelola memori lokal langkah bot (visited_history) dan sinkronisasi memori tim
    ├── combat_handler.py          # Modul pengendali pemrosesan log tempur dan pemutakhiran asinkronus (anti-mati palsu)
    ├── websocket_handler.py       # Handler jaringan WebSocket untuk siklus permainan, matchmaking, dan reset memori
    │
    ├── ai/                        # Folder utama modul kecerdasan buatan agen
    │   ├── __init__.py            # Berkas inisialisasi paket AI
    │   │
    │   ├── detector/              # Subsistem pengurai status lingkungan dari masukan mentah server
    │   │   ├── __init__.py        # Berkas inisialisasi paket detector
    │   │   ├── deadzone_detector.py # Pengurai status wilayah deadzone aktif dan pending collapse
    │   │   ├── enemy_detector.py  # Pengurai status, jarak, dan statistik musuh atau monster secara live
    │   │   ├── loot_detector.py   # Pengurai barang-barang yang tergeletak di lantai sekitar wilayah agen
    │   │   ├── radar_detector.py  # Pengurai peta radar regional sekitar wilayah agen
    │   │   ├── self_detector.py   # Pengurai statistik diri agen seperti HP, EP, inventaris, dan peralatan
    │   │   └── zone_detector.py   # Pengurai wilayah agen, cuaca, jenis terrain, serta gas/radiasi berbahaya
    │   │
    │   └── strategy/              # Subsistem pengambil keputusan strategis agen
    │       ├── __init__.py        # Berkas inisialisasi paket strategi
    │       ├── decision_maker.py  # Mesin utama pemicu Gerbang Filter berbasis Konteks (Emergency vs Combat vs Exploration)
    │       ├── navigation_strategy.py # Logika penentu arah gerakan berbasis rute global dan penarik magnet tim (Leader-Follower)
    │       ├── ruin_exploration_strategy.py # Logika khusus pencarian/eksplorasi barang di daerah ruins
    │       ├── whisper_sabotage.py
    │       │
    │       └── priority/          # Subsistem pembobotan nilai aksi berdasarkan kebutuhan mendesak
    │           ├── __init__.py    # Berkas inisialisasi paket priority
    │           ├── loot_helper.py
    │           ├── equipped_priority.py # Perhitungan nilai urgensi pemasangan senjata melee/ranged adaptif sesuai jarak musuh
    │           ├── ground_loot_priority.py # Perhitungan nilai urgensi pengambilan barang dan bagi-pakai armor cadangan tim
    │           ├── interact_priority.py # Perhitungan prioritas interaksi dengan fasilitas wilayah (Medical, dll)
    │           ├── survival_priority.py # Perhitungan kriteria kelesuan tempur (TTK PvP) dan kabur aman saat HP kritis
    │           ├── recovery_priority.py # Perhitungan prioritas penggunaan ramuan HP/EP atau REST proaktif saat aman
    │           └── target_kill_priority.py # Perhitungan prioritas penyerangan target dengan bonus pengeroyokan (Focus-Fire)
    │
    └── utils/                     # Folder berisi modul utilitas/helper operasional fungsional
        ├── __init__.py            # Berkas inisialisasi paket utils
        ├── action_helper.py       # Generator pembentuk payload JSON pesan aksi (bergerak, serang, discard-drop, dll)
        └── zone_helper.py         # Mesin navigasi penghitung rute terpendek di peta arena
